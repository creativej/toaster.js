Toaster.js demo

Basic usage

var toaster = window.toaster();

var toast = window.toaster.toast({
content: $('.toast-content-input').val()
});
toaster.push(toast);

Toast your text

Toast your html

Just specify your own css for custom styles 

.toast-image {
	padding: 5px 5px 0 5px;
	float: left;
}

Toast can popup in all four corners

var toaster = window.toaster({
	corner: 'bottom-left-corner' // Or 'top-left-corner', 'top-right-corner'
});
